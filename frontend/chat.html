# آلة حاسبة بسيطة
def calculator():
    while True:
        print("\n1. جمع")
        print("2. طرح")
        print("3. ضرب")
        print("4. قسمة")
        print("5. خروج")
        
        choice = input("اختر العملية: ")
        
        if choice == '5':
            break
            
        num1 = float(input("الرقم الأول: "))
        num2 = float(input("الرقم الثاني: "))
        
        if choice == '1':
            print(f"النتيجة: {num1 + num2}")
        elif choice == '2':
            print(f"النتيجة: {num1 - num2}")
        elif choice == '3':
            print(f"النتيجة: {num1 * num2}")
        elif choice == '4':
            if num2 != 0:
                print(f"النتيجة: {num1 / num2}")
            else:
                print("خطأ: القسمة على صفر")

if __name__ == "__main__":
    calculator()
